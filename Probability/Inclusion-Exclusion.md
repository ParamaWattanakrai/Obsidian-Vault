If $A\cup B\neq\emptyset$ then
$$
\begin{flalign}
P(A+B)=P(A)+P(B)-P(AB)&&
\end{flalign}
$$

$$
\begin{flalign}
P(A+B+C)=P(A)+P(B)+P(C)-P(AB)-P(AC)-P(BC)+P(ABC)&&
\end{flalign}
$$
