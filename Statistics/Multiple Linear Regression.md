\begin{flalign}
\hat{y}=a+bx_1+cx_2&&
\end{flalign}
$$
$$
\begin{flalign}
na+b\sum x_1+c\sum x_2=\sum y&&
\end{flalign}
$$
$$
\begin{flalign}
a\sum x_1+b\sum x_1^2+c\sum x_1x_2=\sum x_1y&&
\end{flalign}
$$
$$
\begin{flalign}
a\sum x_2+b\sum x_1x_2+c\sum x_2^2=\sum x_2y&&
\end{flalign}
